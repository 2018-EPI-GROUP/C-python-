## 一.单链表

#### 1.结点定义

```c
typedef struct node 
{
    int data;				//数据域
    struct node *next;		//指针域
}Lnode,*Linlist;
```

#### 2.建立单链表

```c
Linlist Creat-list()
{
    Linlist H=(Linlist)malloc(sizeof(Lnode));
    H->next=NULL;
    Lnode *s;
    int rdata;
    scanf("%d",&rdata);
    while(rdata!=-1)
    {
        s=(Linlist)malloc(sizeof(Lnode));
        s->data=rdata;
        s->next=H->next;
        H->next=s;
        scanf("%d",&rdata);
    }
    return H;
}
```

#### 3.查找单链表数据

```c
int Find-list(Linlist H,int k)	//按数据查找
{
    Lnode *p=H->next;
    while(p!=NULL&&p->data!=k)
    {
        p=p->next;
    }
    return p;
}
```

```c
int get-list(Linlist H,int i)
{
    Lnode *p=H;
    int j=0;
    while(p->next!=NULL&&j<i)
    {
        p=p->next;
        j++;
    }
    if(j==k)
        return p;
    else 
        return NULL;
}
```

#### 6.单链表逆置

```c
void Reverse(Linklist H)
{
    Lnode *p，*q;
    p=H->next;
    H->next=NULL;
    while(p)
    {
        q=p;
        p=p->next;
        q->next=H->next;
        H->next=q;
    }
}
```
