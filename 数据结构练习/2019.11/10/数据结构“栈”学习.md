### 一.数组栈

```c
#define MAXSIZE 100
/*定义结构体*/
typedef struct
{
    int elem[MAXSIZE];				//栈的元素个数
    int top;						//栈顶
}Seqstack;

/*建立栈空间，初始化栈顶指针*/
Seqstack *Init()
{
    Seqstack *s;
    s=(Seqstack *)malloc(sizeof(Seqstack));
    s->top=-1;
    return s;
}

/*判断是否为空栈*/
int Empty(Seqstack *s)
{
    if(s->top==-1)
    	return 1;
    else 
        return 0;
}

/*入栈操作*/
int Push(Seqstack *s,int x)				//将x元素入栈
{
    if(s->top==MAXSIZE-1)
    	return 0;						//栈满不能入栈
    else
    {
        s->top++;
        s->elem[s->top]=x;				//给该位置栈赋值
        return 1;					
    }
}

/*出栈操作*/
int Pop(Seqstack *s,int *x)				//将栈顶元素存入x中
{
    if(Empty(s))
        return 0;
    else
    {
        * x=s->elem[s->top];			
        s->top--;
        return 1;
    }
}

/*取出栈顶元素*/
int Gettop(Seqstack *s)
{
    if(Empty(s))
        return 0;
    else
        return (s->elem[s->top]);
}
```
