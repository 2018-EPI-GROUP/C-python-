/*
顺序表的学习：
1.自定义顺序表（包含head,size,len)
2.定义一个函数initTable()(申请失败自动退出，初始化顺序表，返回顺序表)
3.定义一个displayTable()函数（遍历顺序表，打印其中的值）
4.主函数遍历Size并存入顺序表
*/
#include<stdio.h>
#include<stdlib.h>
#define Size 5
typedef struct Table {
	int* head;
	int len;
	int size;
}table;
table initTable() 
{
	table t;
	t.head = (int*)malloc(Size * sizeof(int));
	if (!t.head)
	{
		printf("申请失败自动退出");
		exit(0);
	}
	t.len = 0;
	t.size = Size;
	return t;
}
