```python
# 导入相关库和文件

# 三层神经网络的实现模型
def model(X, Y, learning_rate = 0.01, num_iterations = 15000, print_cost = True, initialization = "he"):
    """
    实现了一个三层的神经网络:LINEAR->RELU->LINEAR->RELU->LINEAR->SIGMOID
    参数:
    X -- 形状输入数据(2, number of examples)
    Y -- 真正的“标签”向量 (containing 0 for red dots; 1 for blue dots), of shape (1, number of examples)
    learning_rate -- 梯度下降的学习率
    num_iteration -- 运行梯度下降的迭代次数
    print_cost -- 如果为真，则每1000次迭代打印一次成本
    initialization -- 标记选择使用哪个初始化("zeros","random" or "he")
    
    返回:
    parameters -- 从模型中学习的参数 
    """
        
    grads = {}
    costs = [] # 记录损失
    m = X.shape[1]
    layers_dims = [X.shape[0], 10, 5, 1]
    
    # 初始化字典参数
    if initialization == "zeros":
        parameters = initialize_parameters_zeros(layers_dims)
    elif initialization == "random":
        parameters = initialize_parameters_random(layers_dims)
    elif initialization == "he":
        parameters = initialize_parameters_he(layers_dims)

    # 循环(梯度下降)

    for i in range(0, num_iterations):

        # 正向传播
        a3, cache = forward_propagation(X, parameters)
        
        # Loss
        cost = compute_loss(a3, Y)

        # 反向传播
        grads = backward_propagation(X, Y, cache)
        
        # 更新参数
        parameters = update_parameters(parameters, grads, learning_rate)
        
        # 每1000次迭代打印损失
        if print_cost and i % 1000 == 0:
            print("迭代后成本 {}: {}".format(i, cost))
            costs.append(cost)
            
    # 损失图表
    plt.plot(costs)
    plt.ylabel('成本')
    plt.xlabel('迭代(每百)')
    plt.title("学习率 =" + str(learning_rate))
    plt.show()
    
    return parameters
    
# 零值初始化
def initialize_parameters_zeros(layers_dims):
    """
    参数:
    layer_dims -- 包含每个层的大小的python数组(list)。
    
    返回:
    parameters -- python字典包含你的参数"W1", "b1", ..., "WL", "bL":
    W1 -- 形状的权矩阵(layers_dims[1]， layers_dims[0])
    b1 -- 形状的偏置向量(layers_dims[1]， 1)
    """
    
    parameters = {}
    L = len(layers_dims)            # 网络中的层数
    for l in range(1, L):
        parameters['W' + str(l)] = np.zeros((layers_dims[l],layers_dims[l-1]))
        parameters['b' + str(l)] = np.zeros((layers_dims[l],1))
    return parameters
    
# 随机初始化
def initialize_parameters_random(layers_dims):
    """
    参数:
    parameters -- 包含每个层的大小的python数组(list)。
    
    返回:
    parameters -- python字典包含你的参数"W1", "b1", ..., "WL", "bL":
    W1 -- 形状的权矩阵(layers_dims[1]， layers_dims[0])
    b1 -- 形状的偏置向量(layers_dims[1]， 1)
    …
    WL -- 形状的权矩阵(layers_dims[L]， layers_dims[L-1])
    bL -- 形状的偏置向量(layers_dims[L]， 1)
    """
    
    np.random.seed(3)               # 这是一个确定的随机数
    parameters = {}
    L = len(layers_dims)            # 表示层数的整数
    
    for l in range(1, L):
        parameters['W' + str(l)] = np.random.randn(layers_dims[l],layers_dims[l-1])*10
        parameters['b' + str(l)] = np.zeros((layers_dims[l],1))

    return parameters
    
# He 初始化
def initialize_parameters_he(layers_dims):
    """
    参数:
    parameters -- 包含每个层的大小的python数组(列表)。
    
    返回:
    parameters -- python字典包含你的参数"W1", "b1", ..., "WL", "bL":
    W1 -- 形状的权矩阵(layers_dims[1]， layers_dims[0])
    b1 -- 形状的偏置向量(layers_dims[1]， 1)
    …
    WL -- 形状的权矩阵(layers_dims[L]， layers_dims[L-1])
    bL -- 形状的偏置向量(layers_dims[L]， 1)
    """
    
    np.random.seed(3)
    parameters = {}
    L = len(layers_dims) - 1 # 表示层数的整数
     
    for l in range(1, L + 1):
        parameters['W' + str(l)] = np.random.randn(layers_dims[l],layers_dims[l-1])*np.sqrt(2./layers_dims[l-1])
        parameters['b' + str(l)] = np.zeros((layers_dims[l],1))
        
    return parameters
```

```python
# 导入库和文件
import numpy as np
import matplotlib.pyplot as plt
from reg_utils import sigmoid, relu, plot_decision_boundary, initialize_parameters, load_2D_dataset, predict_dec
from reg_utils import compute_cost, predict, forward_propagation, backward_propagation, update_parameters
import sklearn
import sklearn.datasets
import scipy.io
from testCases import *

%matplotlib inline
plt.rcParams['figure.figsize'] = (7.0, 4.0) # 设置图的默认大小
plt.rcParams['image.interpolation'] = 'nearest'
plt.rcParams['image.cmap'] = 'gray'

# 导入数据
train_X, train_Y, test_X, test_Y = load_2D_dataset()

# 未正则化的模型
def model(X, Y, learning_rate = 0.3, num_iterations = 30000, print_cost = True, lambd = 0, keep_prob = 1):
    """
    实现一个三层神经网络: LINEAR->RELU->LINEAR->RELU->LINEAR->SIGMOID.
    参数:
    X -- 输入数据，形状(input size, number of examples)
    Y -- 真正的“标签”向量(1 for blue dot / 0 for red dot), 形状 (output size, number of examples)
    learning_rate -- 优化的学习率
    num_iteration -- 优化循环的迭代次数
    print_cost -- 如果为真，则每10000次迭代打印一次成本
    lambd -- 正则化超参数，标量
    keep_prob -- 神经元在退出时保持活动的概率，标量。
    
    返回:
    parameters -- 从模型中学习的参数。然后他们可以用来预测。

    """
        
    grads = {}
    costs = []                            # 跟踪成本
    m = X.shape[1]                        # 例子
    layers_dims = [X.shape[0], 20, 3, 1]
    
    # 初始化字典参数
    parameters = initialize_parameters(layers_dims)

    # 循环(梯度下降)

    for i in range(0, num_iterations):

        # 向前传播: LINEAR -> RELU -> LINEAR -> RELU -> LINEAR -> SIGMOID.
        if keep_prob == 1:
            a3, cache = forward_propagation(X, parameters)
        elif keep_prob < 1:
            a3, cache = forward_propagation_with_dropout(X, parameters, keep_prob)
        
        # 成本函数
        if lambd == 0:
            cost = compute_cost(a3, Y)
        else:
            cost = compute_cost_with_regularization(a3, Y, parameters, lambd)
            
        # 向后传播.
        assert(lambd==0 or keep_prob==1)    # 可以同时使用L2正则化和dropout 
                                            # 但是这里一次只会探索一个
        if lambd == 0 and keep_prob == 1:
            grads = backward_propagation(X, Y, cache)
        elif lambd != 0:
            grads = backward_propagation_with_regularization(X, Y, cache, lambd)
        elif keep_prob < 1:
            grads = backward_propagation_with_dropout(X, Y, cache, keep_prob)
        
        # 更新参数
        parameters = update_parameters(parameters, grads, learning_rate)
        
        # 每10000次迭代打印一次损失
        if print_cost and i % 10000 == 0:
            print("迭代后成本 {}: {}".format(i, cost))
        if print_cost and i % 1000 == 0:
            costs.append(cost)
    
    # 打印图像
    plt.plot(costs)
    plt.ylabel('成本')
    plt.xlabel('迭代 (x1,000)')
    plt.title("学习率 =" + str(learning_rate))
    plt.show()
    
    return parameters
    
# L2正则化
def compute_cost_with_regularization(A3, Y, parameters, lambd):
    """
    用L2正则化实现成本函数。
    参数:
    A3 -- 后激活，正向传播输出，形状(output size, number of examples)
    Y -- “true”标签向量，形状 (output size, number of examples)
    parameters -- 包含模型参数的python字典
    
    返回:
    cost -- 正则化损失函数的值(formula(2))
    """
    m = Y.shape[1]
    W1 = parameters["W1"]
    W2 = parameters["W2"]
    W3 = parameters["W3"]
    
    cross_entropy_cost = compute_cost(A3, Y) # 得到了成本的交叉熵部分
    
    L2_regularization_cost = (1./m*lambd/2)*(np.sum(np.square(W1)) + np.sum(np.square(W2)) + np.sum(np.square(W3)))
    
    cost = cross_entropy_cost + L2_regularization_cost
    
    return cost
    
# 带正则化的反向传播函数
def backward_propagation_with_regularization(X, Y, cache, lambd):
    """
    实现基线模型的反向传播，我们向其添加了L2正则化。
    
    参数:
    X -- 输入数据集，形状(input size, number of examples)
    Y -- “true”标签向量，形状 (output size, number of examples)
    cache -- 来自forward_propagation()的缓存输出
    lambd -- 正则化超参数，标量
    
    返回:
    gradients -- 一个字典，包含了每个参数、激活和预激活变量的梯度
    """
    
    m = X.shape[1]
    (Z1, A1, W1, b1, Z2, A2, W2, b2, Z3, A3, W3, b3) = cache
    
    dZ3 = A3 - Y
    
    dW3 = 1./m * np.dot(dZ3, A2.T) + lambd/m * W3
    db3 = 1./m * np.sum(dZ3, axis=1, keepdims = True)
    
    dA2 = np.dot(W3.T, dZ3)
    dZ2 = np.multiply(dA2, np.int64(A2 > 0))
    dW2 = 1./m * np.dot(dZ2, A1.T) + lambd/m * W2
    db2 = 1./m * np.sum(dZ2, axis=1, keepdims = True)
    
    dA1 = np.dot(W2.T, dZ2)
    dZ1 = np.multiply(dA1, np.int64(A1 > 0))
    dW1 = 1./m * np.dot(dZ1, X.T) + lambd/m * W1
    db1 = 1./m * np.sum(dZ1, axis=1, keepdims = True)
    
    gradients = {"dZ3": dZ3, "dW3": dW3, "db3": db3,"dA2": dA2,
                 "dZ2": dZ2, "dW2": dW2, "db2": db2, "dA1": dA1, 
                 "dZ1": dZ1, "dW1": dW1, "db1": db1}
    
    return gradients
    
# 前向传播与dropout
def forward_propagation_with_dropout(X, parameters, keep_prob = 0.5):
    """
    实现正向传播: LINEAR -> RELU + DROPOUT -> LINEAR -> RELU + DROPOUT -> LINEAR -> SIGMOID.
    参数:
    X -- 形状的输入数据集(2，样本数量)
    parameters -- python字典，包含你的参  数“W1”，“b1”，“W2”，“b2”，“W3”，“b3”:
    W1 -- 形状的权值矩阵(20,2)
    b1 -- 形状的偏置向量(20,1)
    W2 -- 形状的权值矩阵(3,20)
    b2 -- 形状的偏置向量(3,1)
    W3 -- 形状的权值矩阵(1,3)
    b3 -- 形状的偏置向量(1,1)
    keep_prob -- 神经元在退出时保持活动的概率，标量
    """
    
    np.random.seed(1)
    
    # 检索参数
    W1 = parameters["W1"]
    b1 = parameters["b1"]
    W2 = parameters["W2"]
    b2 = parameters["b2"]
    W3 = parameters["W3"]
    b3 = parameters["b3"]
    
    # LINEAR -> RELU -> LINEAR -> RELU -> LINEAR -> SIGMOID
    Z1 = np.dot(W1, X) + b1
    A1 = relu(Z1)
 
    D1 = np.random.rand(A1.shape[0],A1.shape[1]) 
    D1 = D1 < keep_prob
    A1 = A1 * D1 
    A1 = A1 / keep_prob       

    Z2 = np.dot(W2, A1) + b2
    A2 = relu(Z2)

    D2 = np.random.rand(A2.shape[0],A2.shape[1])  
    D2 = D2 < keep_prob                      
    A2 = A2 * D2                        
    A2 = A2 / keep_prob                     

    Z3 = np.dot(W3, A2) + b3
    A3 = sigmoid(Z3)
    
    cache = (Z1, D1, A1, W1, b1, Z2, D2, A2, W2, b2, Z3, A3, W3, b3)
    
    return A3, cache
    
# 反向传播与dropout
def backward_propagation_with_dropout(X, Y, cache, keep_prob):
    """
    实现基线模型的反向传播，我们将dropout添加到基线模型中。
    
    参数:
    X -- 形状的输入数据集(2, number of examples)
    Y -- “true”标签向量，形状(output size, number of examples)
    缓存 -- 缓存来自forward_propagation_with_dropout()的输出
    keep_prob -- 神经元在退出时保持活动的概率，标量
    
    返回:
     -- gradients -- 一个字典，包含了每个参数、激活和预激活变量的梯度
    """
    
    m = X.shape[1]
    (Z1, D1, A1, W1, b1, Z2, D2, A2, W2, b2, Z3, A3, W3, b3) = cache
    
    dZ3 = A3 - Y
    dW3 = 1./m * np.dot(dZ3, A2.T)
    db3 = 1./m * np.sum(dZ3, axis=1, keepdims = True)
    dA2 = np.dot(W3.T, dZ3)
    dA2 = dA2 * D2   
    dA2 = dA2 / keep_prob 
    dZ2 = np.multiply(dA2, np.int64(A2 > 0))
    dW2 = 1./m * np.dot(dZ2, A1.T)
    db2 = 1./m * np.sum(dZ2, axis=1, keepdims = True)
    
    dA1 = np.dot(W2.T, dZ2)
    dA1 = dA1 * D1    
    dA1 = dA1 / keep_prob   

    dZ1 = np.multiply(dA1, np.int64(A1 > 0))
    dW1 = 1./m * np.dot(dZ1, X.T)
    db1 = 1./m * np.sum(dZ1, axis=1, keepdims = True)
    
    gradients = {"dZ3": dZ3, "dW3": dW3, "db3": db3,"dA2": dA2,
                 "dZ2": dZ2, "dW2": dW2, "db2": db2, "dA1": dA1, 
                 "dZ1": dZ1, "dW1": dW1, "db1": db1}
    
    return gradients
```
